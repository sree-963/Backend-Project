// Stack

class Node{
            constructor(value){
                this.value=value;
                this.next=null;
            }
         }

         class Stack{
            constructor(value){
                const newNode=new Node(value)
                this.top=newNode;
                this.length=1
            }

// Inserting a Value to a Stack

            push(value){
                const newNode=new Node(value)
                if(!this.top) this.top=newNode

                newNode.next=this.top
                this.top=newNode
                this.length++
                return this;
            }

// Deleting A value from the Stack

            pop(){
                if(this.length===0) 
                return console.log('Stack is Empty')
                let temp=this.top;
                this.top=this.top.next;
                temp.next=null;
                this.length--
                return temp;
            }
         }
 
// Stack Calling

         let s=new Stack(3)
         s.push(4)
        s.push(5)
        s.push(6)
        s.pop()
        console.log(s)
